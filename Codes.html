prepPageList <- function(ii) {
  generate_url <- function(url){
    output <- paste("https://www.sahibinden.com", url, sep="")
    return(output)
  }
  cat("Page : ", ii, "\n")
  durl <- paste0("https://www.sahibinden.com/satilik-daire/izmir-karsiyaka-nergiz?pagingOffset=",((ii+5)*20),"&a20=38470&a20=38474")
  doc <- read_html(durl)
  doc_ilanlar <- html_nodes(doc, "a.classifiedTitle")
  ilan_baslik<- doc_ilanlar %>%
    html_text("tittle") 
  ilan_linkleri<- doc_ilanlar %>%
    html_attr("href") 
  son_linkler <- lapply(ilan_linkleri, generate_url)
  doc_fiyat <- html_nodes(doc, "div.classified-price-container")
  fiyat1 <- html_text2(doc_fiyat)
  c <- data.table(
    urls = son_linkler,
    baslik = ilan_baslik,
    fiyat = fiyat1
  )
  Sys.sleep(3)
  return(c)
}
nPage<- 1
Linkler <- lapply(c(1:nPage), prepPageList)
tumLinkler <- do.call("rbind", Linkler)
